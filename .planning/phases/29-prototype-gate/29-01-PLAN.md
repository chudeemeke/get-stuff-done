```javascript
function createTempDir() {
  // Returns { path: string, cleanup: function }
  const tmpPath = fs.mkdtempSync(path.join(os.tmpdir(), 'gsd-test-'));
  const cleanup = () => {
    if (fs.existsSync(tmpPath)) {
      fs.rmSync(tmpPath, { recursive: true, force: true });
    }
  };
  return { path: tmpPath, cleanup };
}
```

From tests/installer.test.js (established subprocess pattern):
```javascript
function runInstaller(env = {}, args = []) {
  try {
    const result = execSync(`node "${INSTALL_SCRIPT}" ${args.join(' ')}`, {
      encoding: 'utf-8',
      env: { ...process.env, ...env },
      stdio: ['pipe', 'pipe', 'pipe'],
    });
    return { success: true, output: result };
  } catch (err) {
    return {
      success: false,
      output: err.stdout?.toString() || '',
      error: err.stderr?.toString() || err.message,
    };
  }
}
```

         - Runs via `execSync(`node "${installScript}" ${args.join(' ')}`, { encoding: 'utf-8', env: { ...process.env }, stdio: ['pipe', 'pipe', 'pipe'], timeout: 30000 })`

         - Reads `scratchDir/bin/install.js` content
         - Applies ONLY these substitutions (per research -- exact npm package name strings only):
           - `"get-shit-done-cc"` -> `"@chude/get-stuff-done"` (note: only the npm package name with -cc suffix)
           - `"npx get-shit-done-cc"` -> `"bunx @chude/get-stuff-done"` (install command examples)
           - `"get-shit-done-cc@latest"` -> `"@chude/get-stuff-done@latest"` (version-pinned references)
         - CRITICAL: Does NOT replace bare `"get-shit-done"` (without `-cc`) -- this appears 130+ times in path resolution

       - Helper function `copyOverlayAdditions(targetDir)`:
         - Creates placeholder overlay files (these are NOT real fork code -- just test fixtures proving the mechanism):
