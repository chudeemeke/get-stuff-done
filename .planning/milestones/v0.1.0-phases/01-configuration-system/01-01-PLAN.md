function getConfigValue(config, path, defaultValue) {
  const parts = path.split('.');
  let value = config;
  for (const part of parts) {

      return defaultValue;
    }
    value = value[part];
  }
  return value !== undefined ? value : defaultValue;
}

  <name>Task 3b: Replace jq with Node.js ConfigLoader in bin/gsd</name>
  <files>bin/gsd</files>
  <action>
Replace the jq-based read_config() function with Node.js calls to ConfigLoader.

**Current code (using jq):**
```bash
read_config() {
    local key="$1"
    local default="$2"
    local value=""

    # Try project-level config first
    if [[ -f "$GSD_PROJECT_CONFIG" ]]; then
        value=$(jq -r "$key // empty" "$GSD_PROJECT_CONFIG" 2>/dev/null || true)
    fi

    # Fall back to user-level config
    if [[ -z "$value" && -f "$GSD_CONFIG" ]]; then
        value=$(jq -r "$key // empty" "$GSD_CONFIG" 2>/dev/null || true)
    fi

    # Return value or default
    echo "${value:-$default}"
}
```

**Replace with:**
```bash
# Read config value using Node.js ConfigLoader (handles JSON5, validation, defaults)
read_config() {
    local key="$1"
    local default="$2"

    node -e "
        const {loadConfig, getConfigValue} = require('${GSD_INSTALL_DIR}/src/config/ConfigLoader');
        try {
            const config = loadConfig();
            const val = getConfigValue(config, '${key}', '${default}');
            console.log(val);
        } catch (e) {
            console.error('Config error:', e.message);
            console.log('${default}');
        }
    " 2>/dev/null || echo "$default"
}
```
