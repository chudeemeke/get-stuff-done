```javascript
function getConfigValue(config, path, defaultValue) {
  const parts = path.split('.');
  let value = config;
  for (const part of parts) {
    if (value == null || typeof value !== 'object') {
      return defaultValue;
    }
    value = value[part];
  }
  return value !== undefined ? value : defaultValue;
}
```


Replace with Node.js:
```bash
# Read config using Node.js ConfigLoader
read_config() {
    local key="$1"
    local default="$2"

    # Use Node.js to read config (handles JSON5, validation, defaults)
    local value
    value=$(node -e "
        const {loadConfig, getConfigValue} = require('$GSD_INSTALL_DIR/src/config/ConfigLoader');
        try {
            const config = loadConfig();
            const val = getConfigValue(config, '$key', '$default');
            console.log(val);
        } catch (e) {
            console.error('Config error:', e.message);
            console.log('$default');
        }
    " 2>&1)

    echo "$value"
}
```

```javascript
const yellowMax = autocompactThreshold * 0.75;
const orangeMax = autocompactThreshold * 0.875;
```

```javascript
// Color based on usage (dynamic thresholds)
if (used < greenMax) {
```
