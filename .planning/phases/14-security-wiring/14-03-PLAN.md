**Step 1: Create src/config/schema.js**

Define an AJV schema for `.planning/config.json` based on the structure in gsd-tools.js loadConfig() (lines 58-69 of gsd-tools.js). The config has these top-level keys:
- `model_profile` -- string, one of ['quality', 'balanced', 'budget']
- `commit_docs` -- boolean
- `search_gitignored` -- boolean
- `branching_strategy` -- string, one of ['none', 'phase', 'milestone']
- `phase_branch_template` -- string
- `milestone_branch_template` -- string
- `research` -- boolean
- `plan_checker` -- boolean
- `verifier` -- boolean
- `parallelization` -- boolean or object with `enabled` field

BUT: gsd-tools.js loadConfig() also supports NESTED format (the `get()` helper checks `parsed[nested.section][nested.field]`). So the actual config.json can have:
```json
{
  "model_profile": "balanced",
  "planning": { "commit_docs": true, "search_gitignored": false },
  "git": { "branching_strategy": "none", "phase_branch_template": "...", "milestone_branch_template": "..." },
  "workflow": { "research": true, "plan_check": true, "verifier": true }
}
```

The schema should accept EITHER flat or nested format. Use AJV (already in deps). Export the schema object and a validate function:

```javascript
const Ajv = require('ajv');

const planningConfigSchema = {
  $schema: 'http://json-schema.org/draft-07/schema#',
  type: 'object',
  // Accept both flat keys and nested sections
  properties: {
    model_profile: { type: 'string', enum: ['quality', 'balanced', 'budget'] },
    commit_docs: { type: 'boolean' },
    search_gitignored: { type: 'boolean' },
    branching_strategy: { type: 'string', enum: ['none', 'phase', 'milestone'] },
    phase_branch_template: { type: 'string' },
    milestone_branch_template: { type: 'string' },
    research: { type: 'boolean' },
    plan_checker: { type: 'boolean' },
    verifier: { type: 'boolean' },
    parallelization: {
      oneOf: [
        { type: 'boolean' },
        { type: 'object', properties: { enabled: { type: 'boolean' } }, required: ['enabled'] }
      ]
    },
    // Nested sections
    planning: {
      type: 'object',
      properties: {
        commit_docs: { type: 'boolean' },
        search_gitignored: { type: 'boolean' }
      },
      additionalProperties: false
    },
    git: {
      type: 'object',
      properties: {
        branching_strategy: { type: 'string', enum: ['none', 'phase', 'milestone'] },
        phase_branch_template: { type: 'string' },
        milestone_branch_template: { type: 'string' }
      },
      additionalProperties: false
    },
    workflow: {
      type: 'object',
      properties: {
        research: { type: 'boolean' },
        plan_check: { type: 'boolean' },
        verifier: { type: 'boolean' }
      },
      additionalProperties: false
    },
    // Memory, effort, teams from Phase 10
    memory: { type: 'object' },
    effort: { type: 'object' },
    teams: { type: 'object' }
  },
  additionalProperties: false
};

const ajv = new Ajv({ allErrors: true });
const validate = ajv.compile(planningConfigSchema);

function validatePlanningConfig(config) {
  const valid = validate(config);
  if (!valid) {
    return { ok: false, errors: validate.errors.map(e => e.message || `${e.keyword}: ${JSON.stringify(e.params)}`) };
  }
  return { ok: true };
}

module.exports = { planningConfigSchema, validatePlanningConfig };
```

Read the actual `.planning/config.json` file in the project to ensure the schema covers its real keys. The schema above may need adjustment based on what keys actually exist (memory, effort, teams were added in Phase 10).

**Step 2: Create bin/validate-configs.js**

Standalone script (NOT a bin entry, just in the files array). Structure:

```javascript
#!/usr/bin/env node
'use strict';

// Escape hatch (CONTEXT locked decision)
if (process.env.SKIP_CONFIG_VALIDATION === '1') {
  process.stderr.write('Warning: Config validation skipped (SKIP_CONFIG_VALIDATION=1)\n');
  process.exit(0);
}

const fs = require('fs');
const path = require('path');

const PROJECT_ROOT = path.resolve(__dirname, '..');

// Validator registry
const validators = {
  '.planning/config.json': validatePlanningConfig,
  '.planning/ROADMAP.md': validateMarkdownSections('ROADMAP', ['## Milestones', '## Phases', '## Progress']),
  '.planning/STATE.md': validateMarkdownSections('STATE', ['## Current Position', '## Session Continuity']),
  '.planning/PROJECT.md': validateMarkdownSections('PROJECT', ['## What This Is', '## Requirements', '## Context']),
  'package.json': validatePackageJson,
};
```

Implement each validator:
1. **validatePlanningConfig:** Read file, JSON.parse, call validatePlanningConfig from schema.js. Check for conflict markers.
2. **validateMarkdownSections(name, requiredSections):** Returns a function that reads the markdown file and checks each required section exists via regex. Also check for conflict markers (`<<<<<<<`, `=======`, `>>>>>>>`).
3. **validatePackageJson:** Read and parse package.json. Verify `files` array entries exist on disk (reuse logic similar to scripts/check-parity.js). Check for conflict markers.

Each validator returns `{ ok: true }` or `{ ok: false, errors: ['message1', ...'] }`.

Output format:
- Per-file: `PASS file` to stdout, or `FAIL file: error1, error2` to stderr
- Summary line: `N passed, M failed` to stdout
- Exit 0 if all pass, exit 1 if any fail
- Support `--quiet` flag: only print failures and summary

The script should gracefully handle missing files (e.g., if .planning/ does not exist, that is a FAIL for those validators, not a crash).
